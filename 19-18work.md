### 1. `asyncHandler.ts` - Import and Type Export Fix
**File:** `backend/src/utils/asyncHandler.ts`
**Date:** 2026-05-19

**Problem:**
- Module had no default export
- Type conflict between Express Request and global fetch API Request
- Return type was too strict (`Promise<void>`)

**Solution:**
```typescript
// BEFORE (broken)
const asyncHandler = (fn: AsyncFunction) => { ... };
export default asyncHandler;

// AFTER (fixed)
import type { Request, Response, NextFunction } from 'express';

type AsyncFunction = (
  req: Request,
  res: Response,
  next: NextFunction
) => Promise<unknown>;

export function asyncHandler(
  fn: (req: Request, res: Response, next: NextFunction) => Promise<unknown>
): (req: Request, res: Response, next: NextFunction) => void {
  return (req: Request, res: Response, next: NextFunction) => {
    Promise.resolve(fn(req, res, next)).catch(next);
  };
}
```

**Changes:**
- Changed from default export to named export
- Used `import type` for explicit Express types
- Changed return type from `Promise<void>` to `Promise<unknown>` to accept handlers returning Response

**File:** `backend/src/services/alert.service.ts`
**Date:** 2026-05-19

**Problem:**

**Solution:**
Added transformer method:

class AlertService {
  private toAlert(doc: mongoose.Document): Alert {
    const obj = doc.toObject();
    return {
      id: obj._id?.toString() || '',
      type: obj.type,
      severity: obj.severity,
      message: obj.message,
      metadata: obj.metadata,
      acknowledged: obj.acknowledged,
      acknowledgedBy: obj.acknowledgedBy,
      acknowledgedAt: obj.acknowledgedAt,
      resolved: obj.resolved,
      resolvedBy: obj.resolvedBy,
      resolvedAt: obj.resolvedAt,
      createdAt: obj.createdAt,
    };
  }

  async create(type, severity, message, metadata?): Promise<Alert> {
    const alert = new AlertModel({ type, severity, message, metadata });
    await alert.save();
    console.log(`[ALERT] ${severity.toUpperCase()} - ${message}`);
    return this.toAlert(alert);  // Changed from alert.toObject()
  }

  async getActive(): Promise<Alert[]> {
    const docs = await AlertModel.find({ resolved: false }).sort({ createdAt: -1 });
    return docs.map((doc) => this.toAlert(doc));  // Map to Alert[]
  }
  // ... similar for getBySeverity
}
